%TREANGL
gsave
%!PS-Adobe-1.0
 
/infoFile (1.14 FAH ) def
            %currentpagedevice /PageSize get == flush  %% stack has array [width height]
            %-500 -500 moveto 
            %-2323 -327 translate
           170 99 translate

			0 0 moveto 

	% 0 setgray
	% /DeviceRGB setcolorspace
    % 0.96 0.96 0.96 setcolor
	% fill
72 25.4 div	% 1 мм = 72/25.4 пунктов
dup		% дублировать значение на вершине стека
/mm 72 25.4 div  def
/rect 5 mm mul def
/movadd 0.5 mm mul def
/movsub movadd neg def
        

/pagewidth currentpagedevice /PageSize get 0 get def
/pageheight currentpagedevice /PageSize get 1 get def

/mmh pageheight 8 sub def

/W pagewidth movadd sub def
/H pageheight movadd sub def
/Ws pagewidth movadd add def
/Hs pageheight movadd add def

/text {
	/Helvetica findfont	% взять шрифт Helvetica
	2 scalefont	% растянуть до размера 8 (у нас - мм!)
	setfont		% установить выбранный шрифт
	scale		% растянуть в это количество раз по обеим координатам
  	0 movadd moveto	% перейти в точку (0мм, 0мм)
	    	infoFile show	% написать слово текущим шрифтом
       } def
gsave
	rect 0 translate
	text
grestore

/treangl {
newpath
	0 rect moveto
	0 0 lineto
	rect 0 lineto
	0.6 setlinewidth
	stroke
         } def
gsave
	movadd movadd translate
	treangl
grestore
gsave
	H W translate
	180 rotate
	treangl
grestore
gsave
	H movadd translate
	90 rotate
	treangl
grestore
gsave
	movadd W translate
	270 rotate
	treangl
grestore
grestore
%END TREANGL
