%TREANGL
gsave

% По умолчанию, постскрипт использует единицу измерения 1 пункт=1/72 дюйма, 
/infoFile (1.14 FAH ) def

/setpos { % x-abs y-abs  setpos  -
    matrix defaultmatrix transform % x' y'  %to device space
    itransform % x y  %back to current user space
    translate
} def

/rotateto { % angle  rotateto  -
    matrix rotate % create a rotation matrix
    dup 4
    matrix currentmatrix 4 2 getinterval %get the current translation
    putinterval setmatrix % put translation into rot. matrix and install
} bind def

            % currentpagedevice /PageSize get  %% stack has array [width height]
            % 0 0 moveto                       %% start at bottom left

%-90 rotateto
%-90 rotate
0 0 setpos
	0 0 translate
1 1 scale   % or 1 -1 scale for flip  (ViewingOrientation normal 0 1 -1 0 flip ViewingOrientation: 1 0 0 1)
	
	%current transformation matrix 
	% 0 setgray
	% /DeviceRGB setcolorspace
    % 0.96 0.96 0.96 setcolor
	% fill
72 25.4 div	% 1 мм = 72/25.4 пунктов
dup		% дублировать значение на вершине стека
/mm 72 25.4 div  def
/rect 5 mm mul def
/pagewidth currentpagedevice /PageSize get 0 get def
/pageheight currentpagedevice /PageSize get 1 get def
	0 0 translate
/mmh pageheight 8 sub def
/text {
	/Helvetica findfont	% взять шрифт Helvetica
	2 scalefont	% растянуть до размера 8 (у нас - мм!)
	setfont		% установить выбранный шрифт
	scale		% растянуть в это количество раз по обеим координатам
  	0 0 moveto	% перейти в точку (0мм, 0мм)

    	infoFile show	% написать слово текущим шрифтом
       } def
gsave
	rect mmh translate
	text
grestore
/treangl {
	0 rect moveto
	0 0 lineto
	rect 0 lineto
	0.6 setlinewidth
	stroke
              } def
	0 0 translate
	treangl
gsave
	pagewidth pageheight translate
	180 rotate
	treangl
grestore
gsave
	pagewidth 0 translate
	90 rotate
	treangl
grestore
gsave
	0 pageheight translate
	270 rotate
	treangl
grestore
grestore
%END TREANGL
